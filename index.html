function inviaDati() { const modello = document.getElementById('modello').value; const dimensione = document.getElementById('dimensione').value; const capacitaScambio = document.getElementById('capacitaScambio').value; const minutiAcquaSale = document.getElementById('minutiAcquaSale').value; const quantitaSale = document.getElementById('quantitaSale').value; const quantitaResina = document.getElementById('quantitaResina').value; const quantitaSabbia = document.getElementById('quantitaSabbia').value; const coloreUgello = document.getElementById('coloreUgello').value; const coloreInserti = document.getElementById('coloreInserti').value; const dimensioneCollo = document.getElementById('dimensioneCollo').value; const spazioLibero = document.getElementById('spazioLibero').value; if (modello && dimensione && capacitaScambio && minutiAcquaSale && quantitaSale && quantitaResina && quantitaSabbia && spazioLibero && dimensioneCollo && (coloreUgello || modello === "he 1" || modello === "he 1.5")) { db.collection("Addolcitori").doc(modello).collection("dimensioni").doc(dimensione).set({ capacitaScambio: capacitaScambio, minutiAcquaSale: minutiAcquaSale, quantitaSale: quantitaSale, quantitaResina: quantitaResina, quantitaSabbia: quantitaSabbia, coloreUgello: coloreUgello, coloreInserti: coloreInserti, dimensioneCollo: dimensioneCollo, spazioLibero: spazioLibero }) .then(() => { alert("Dati inviati con successo!"); }) .catch((error) => { console.error("Errore durante l'invio dei dati: ", error); }); } else { alert("Per favore, compila tutti i campi richiesti."); } }
